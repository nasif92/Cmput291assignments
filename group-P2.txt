rtext CHAR(20)
rscore INT
email CHAR(25) NOT NULL

rno INT
price Float
rdate CHAR(11)
seats INT

cno INT
PRIMARY KEY(rno)
FOREIGN KEY(cno) REFERENCES Cars)

CREATE TABLE Rides_Offer(
rno INT
price Float
rdate CHAR(11)
seats INT
lugDesc char(25)
licNo CHAR(11) NOT NULL
PRIMARY KEY(rno)
FOREIGN KEY(licNo REFERENCES DRIVERS ON DELETE NO ACTION)
)

CREATE TABLE Rides_dst(
rno INT
price Float
rdate CHAR(11)
seats INT
lugDesc char(25)
lcode NOT NULL
PRIMARY KEY(rno)
FOREIGN KEY(lcode REFERENCES LOCATIONS ON DELETE NO ACTION)
)

CREATE TABLE Bookings_reserve(
bno INT
seats INT

email CHAR(25) NOT NULL
PRIMARY KEY (bno)
FOREIGN KEY (email ON DELETE NO ACTION)
)

CREATE TABLE Bookings_isFor(
bno INT

rno INT NOT NULL

CREATE TABLE Bookings_pickup (
bno INT
seats INT

lcode INT NOT NULL
PRIMARY KEY(bno)
FOREIGN KEY(lcode REFERENCES Locations ON DELETE NO ACTION))

lcode INT
amount FLOAT

email CHAR(25)

CREATE TABLE Locations(
