CREATE TABLE Bookings_Reserve(
 bno INTEGER,
 cost REAL,
 seats INTEGER
 email CHAR(30) NOT NULL,
 name CHAR(30),
 phone INTEGER,
 PRIMARY KEY(bno),
 FOREIGN KEY(email) REFERENCES Members ON DELETE NO ACTION)) 

CREATE TABLE Bookings_for_Rides(
 bno INTEGER,
 cost REAL,
 seats INTEGER,
 rno INTEGER NOT NULL,
 price REAL,
 rdate DATE,
 seats INTEGER,
 lugDesc INTEGER,
 PRIMARY KEY(bno),
 FOREIGN KEY(rno REFERENCES rides ON DELETE NO ACTION))

CREATE TABLE locations(
 bno INTEGER,
 cost REAL,
 seats INTEGER,
 lcode CHAR(20) NOT NULL,
 city CHAR(20),
 prov CHAR(20),
 address CHAR(50),
 PRIMARY KEY(bno,lcode),
 FOREIGN KEY(lcode) REFERENCES Locations,
 FOREIGN KEY(bno) REFERENCES Bookings)

CREATE TABLE Requests(
 amount REAL,
 qdate DATE,
 lcode CHAR(20),
 city CHAR(20),
 prov CHAR(20),
 address CHAR(50),
 email CHAR(30) NOT NULL,
 name CHAR(30),
 phone INTEGER,
 PRIMARY KEY(lcode, email),
 FOREIGN KEY(lcode) REFERENCES Locations,
 FOREIGN KEY(email) REFERENCES Members)

CREATE TABLE enroute(
 lcode CHAR(20),
 city CHAR(20),
 prov CHAR(20),
 address CHAR(50),
 rno INTEGER,
 price REAL,
 rdate DATE,
 seats INTEGER,
 lugDesc INTEGER,
 PRIMARY KEY(rno, lcode),
 FOREIGN KEY(rno) REFERENCES rides,
 FOREIGN KEY(lcode) REFERENCES Locations)

CREATE TABLE Ride_Offer(
 rno INTEGER,
 price REAL,
 rdate DATE,
 seats INTEGER,
 lugDesc INTEGER,
 licNo CHAR(30),
 PRIMARY KEY(rno)
)

CREATE TABLE Ride_Cars_assoc(
 rno INTEGER,
 price REAL,
 rdate DATE,
 seats INTEGER,
 lugDesc INTEGER,
 cno CHAR(20),
 year DATE,
 model CHAR(20),
 make CHAR(20),
 seats INTEGER,
 PRIMARY KEY(rno),
 FOREIGN KEY(cno) REFERENCES Cars)
 
CREATE TABLE Ride_Review_assoc(
 rid CHAR(20),
 rdate DATE,
 rtext CHAR(100),
 rscore INTEGER,
 rno INTEGER,
 price REAL,
 rdate DATE,
 seats INTEGER,
 lugDesc INTEGER,
 PRIMARY KEY(rid),
 FOREIGN KEY(rno) REFERENCES rides)


CREATE TABLE Cars_Register(
 cno CHAR(20),
 year DATE,
 model CHAR(20),
 make CHAR(20),
 seats INTEGER,
 gdate DATE,
 licNo CHAR(30),
 PRIMARY KEY(cno)
)

CREATE TABLE Member_Review(
 rid CHAR(20),
 rdate DATE,
 rtext CHAR(100),
 rscore INTEGER,
 email CHAR(30) NOT NULL,
 name CHAR(30),
 phone INTEGER,
 PRIMARY KEY(rid),
 FOREIGN KEY(email) REFERENCES Members ON DELETE NO ACTION))
